  <script>
    const music = document.getElementById('bgMusic');
    let playing = false;

    // тихий автозапуск
    music.volume = 0.12; // очень тихо

    window.addEventListener('load', () => {
      try {
        music.play();
        playing = true;
      } catch (e) {}
    });

    // при первом касании — снимаем mute
    document.addEventListener('click', () => {
      if (music.muted) {
        music.muted = false; // теперь слышно, но тихо
      }
    }, { once: true });

    function startMusic() {
      if (!playing) {
        music.muted = false;
        music.play();
        playing = true;
      } else {
        music.pause();
        playing = false;
      }
    }

    function sendRSVP() {
      const name = document.getElementById('name').value;
      const withWho = document.getElementById('withWho').value;
      const status = document.getElementById('status').value;

      if (!name || !status) {
        alert('Пожалуйста, введите имя и выберите вариант ответа');
        return;
      }

      const phone = '77075854898';
      const message = `Тұсау кесер Айдай

Имя: ${name}
С кем придёт: ${withWho || 'Не указано'}
Ответ: ${status}`;
      const url = `https://wa.me/${phone}?text=${encodeURIComponent(message)}`;

      window.open(url, '_blank');
      document.getElementById('thanks').style.display = 'block';
    }
  </script>
